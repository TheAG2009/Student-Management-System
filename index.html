🚀 Quick Installation

# Clone the repository
git clone https://github.com/YOUR_USERNAME/Student-Management-System.git
cd Student-Management-System

# Run setup script
chmod +x setup.sh && ./setup.sh # Linux/Mac
setup.bat # Windows

# Compile and run
make clean && make && make run
